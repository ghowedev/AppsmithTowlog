SELECT 
    t.id,
    to_char(t.date, 'MM-DD-YYYY') AS date,        -- formatted for UI
    to_char(t.time, 'HH24MI')      AS time,       -- formatted as HHMM
    t.location,
    t.model,
    t.vin,
    t.licenseplate,
    t.vehicleyear,
    t.notes,
    t.hold,
    t.released,
    makes.make,
    states.state,
    colors.color,
    reasons.reason,
    t.make_id,
    t.color_id,
    t.licensestate_id,
    t.towreason_id,
    t.towcompany_id,
    t.towdate
FROM public.tows_policetow t
JOIN public.enum_makes        makes   ON t.make_id = makes.id
JOIN public.enum_colors       colors  ON t.color_id = colors.id
JOIN public.enum_states       states  ON t.licensestate_id = states.id
JOIN public.enum_towreasons   reasons ON t.towreason_id = reasons.id;
